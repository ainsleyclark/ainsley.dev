{{- $ret := "" -}} {{- $baseURL := getenv "VERCEL_URL" -}} {{- if not $baseURL -}} {{- errorf "VERCEL_URL environment file not found, use vercel dev" -}} {{- end -}} {{ if strings.Contains $baseURL "localhost" -}} {{- $ret = printf "http://%s" $baseURL -}} {{- else -}} {{- $ret = printf "https://%s" $baseURL -}} {{- end -}} {{ return $ret }}
